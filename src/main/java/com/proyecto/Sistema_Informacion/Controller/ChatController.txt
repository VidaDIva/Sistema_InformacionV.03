package com.proyecto.Sistema_Informacion.Controller;

import java.util.ArrayList;
import java.util.List;

import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;

import com.proyecto.Sistema_Informacion.Model.entity.Crear;
import com.proyecto.Sistema_Informacion.Model.service.ExamenMedicoService;
import com.proyecto.Sistema_Informacion.Model.service.IAService;

import jakarta.servlet.http.HttpSession;

@Controller
@RequestMapping("/doctor/chat")
public class ChatController {

    private final IAService iaService;
    private final ExamenMedicoService examenService;

    public ChatController(IAService iaService, ExamenMedicoService examenService) {
        this.iaService = iaService;
        this.examenService = examenService;
    }

    // 🟢 Mostrar chat
    @GetMapping
    public String chat(Model model, HttpSession session) {

        Crear usuario = (Crear) session.getAttribute("usuario");

        if (usuario == null) {
            return "redirect:/login";
        }

        // historial
        List<String> historial = (List<String>) session.getAttribute("chatHistorial");
        if (historial == null) {
            historial = new ArrayList<>();
        }

        model.addAttribute("historial", historial);

        return "Doctor/chat";
    }

    // 🟢 Enviar mensaje
    @PostMapping
    public String enviarMensaje(@RequestParam String mensaje,
                                Model model,
                                HttpSession session) {

        Crear usuario = (Crear) session.getAttribute("usuario");

        if (usuario == null) {
            return "redirect:/login";
        }

        // 🔥 Traer datos reales del médico
        String contexto = "Eres un asistente médico. "
                + "El doctor tiene estos exámenes: "
                + examenService.buscarPorMedico(usuario.getId());

        String respuesta = iaService.preguntarIA(contexto + "\nPregunta: " + mensaje);

        // 🧠 guardar historial
        List<String> historial = (List<String>) session.getAttribute("chatHistorial");
        if (historial == null) {
            historial = new ArrayList<>();
        }

        historial.add("Tú: " + mensaje);
        historial.add("IA: " + respuesta);

        session.setAttribute("chatHistorial", historial);

        model.addAttribute("historial", historial);

        return "Doctor/chat";
    }
}
